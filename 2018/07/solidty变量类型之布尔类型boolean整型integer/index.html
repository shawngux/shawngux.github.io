Solidty变量类型之布尔类型（Boolean）、整型（Integer）
Tuesday, Jul 17, 2018

布尔类型（Booleans）

https://solidity.readthedocs.io/en/latest/types.html#booleans

bool：可能的取值为true和false。

支持的运算符有：

  • !（逻辑非）
  • &&（逻辑与，”and”）
  • ||（逻辑或）
  • ==（等于）
  • !=（不等于）

注：运算符||和&&遵循短路规则。

pragma solidity ^0.4.24;

contract Test {
    uint a;
    uint b;
    bool c;
    bool d;
    
    function Test() {
        a = 1;
        b = 2;
        c = true;
        d = false;
    }
    
    // output: false
    function fnA() constant returns (bool) {
        return !c;
    }
    
    // output: false
    function fnB() constant returns (bool) {
        return c && d;
    }
    
    // output: true
    function fnC() constant returns (bool) {
        return c || d;
    }
    
    // output: false
    function fnD() constant returns (bool) {
        return a == b;
    }
    
    // output: true
    function fnE() constant returns (bool) {
        return a != b;
    }
    
    // 短路效果，output: 1
    function fnF() constant returns (uint) {
        if (c || d) {
            return 1;
        }
        return 0;
    }
}

整型（Integers）

https://solidity.readthedocs.io/en/latest/types.html#integers

整型分为有符号int和无符号uint两种。

uint：支持 uint8 ~ uint256。

int：支持 int8 ~ int256。

默认为 uint256 和 int256。

有符号整型可以表示负数，同时能表示的正数的范围相对于无符号的要缩小。因为其一般的数值要用来表示负数。例如： uint8的范围为：0 ~ 255，int8的范围为：-127 ~ 127。

支持的运算符有：

  • 比较运算符：>、>=、<、<=、==、!=，结果返回布尔值（bool）。
  • 位运算符：&、|、^（异或）、~（取反）。
  • 算术运算符：+、-、x（乘）、/、%（取余）、幂运算、<<（左位移）、>>（右位移）。
pragma solidity ^0.4.24;

contract Test {
    uint8 a = 2;
    uint16 b = 4;
    int c = 8;
    int d = 16;
    
    // 比较运算
    function fnA() constant returns (bool) {
        return a >= c;
    }
    
    // 同类型相加才能得到正确的值
    function fnB() constant returns (uint) {
        return c + b;
    }
    
    // 幂运算
    function fnC() constant returns (uint) {
        return a ** a;
    }
    
    // 位运算
    function fnD() contant returns (int) {
        return d << 2;
    }
    
    // ......
}

注意：在 Solidity 中，右移和除是等价的，因此对一个负数进行右移操作会导致向 0 的取整（截断）。
